```python
list_of_tuples = [("Betty", 45), ("John" , 84), ("Malik" , 34), ("Jose" , 20)]

for pair in list_of_tuples:
    print(pair[0])
    print(pair[1])
```

```python
list_of_tuples = [("Betty", 45), ("John" , 84), ("Malik" , 34), ("Jose" , 20)]

for name, age in list_of_tuples:
    print(name)
    print(age)
```
